! build a one-level RAS with overlap 2 and ILU(0) on the local blocks.
call P%init(’AS’,info)
call P%set(’SUB_OVR’,2,info)
call P%build(A,desc_A,info)
... ...
! solve Ax=b with preconditioned BiCGSTAB
